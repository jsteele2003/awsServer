        function makeContent(location) {
            var contentHolder = '<div id="content">' +
                '<div id="siteNotice">' +
                '</div>';
            contentHolder = contentHolder + '<h2>' + location._id.address.substr(0, location._id.address.indexOf(',')) + '</h2>';
            for (var i = 0; i < location.groups.length; i++) {
            if (i == 0) {
            } 
                contentHolder = contentHolder + '<h1 id="firstHeading" class="firstHeading">';
                contentHolder = contentHolder + location.groups[i].group + '</h1>';
                
                for (var j = 0; j < location.groups[i].type.length; j++) {
                    var type;
                    if (location.groups[i].type[j] == "BB") {
                        type = "Big Book ";
                    }
                    else if (location.meetingGroups[i].meetingType[j] == "B") {
                        type = "Beginners ";
                    }
                    else if (location.meetingGroups[i].meetingType[j] == "C") {
                        type = "Closed Discussion ";
                    }
                    else if (location.meetingGroups[i].meetingType[j] == "S") {
                        type = "Step ";
                    }
                    else if (location.meetingGroups[i].meetingType[j] == "T") {
                        type = "Tradition ";
                    }
                    else if (location.meetingGroups[i].meetingType[j] == "O") {
                        type = "Open Meeting ";
                    }
                    else if (location.meetingGroups[i].meetingType[j] == "OD") {
                        type = "Open Discussion ";
                    }
                    else {
                        type = "";
                    }

                    contentHolder = contentHolder + location.groups[i].time[j] + ' ' + type + '<br>';
                }
                contentHolder = contentHolder + '</p>';
                contentHolder = contentHolder + '<div class="iw-bottom-gradient"></div>';
            }
            return contentHolder;
        }

        function setMarkers(map, locations) {
            // Add markers to the map
            var infowindow = new google.maps.InfoWindow({
                maxWidth: 200
            });

            for (var i = 0; i < locations.length; i++) {
                var location = locations[i];
                var myLatLng = new google.maps.LatLng(location._id.latLong[0], location._id.latLong[1]);

                var contentWindow = makeContent(locations[i]);


                var marker = new google.maps.Marker({
                    position: myLatLng,
                    map: map,
                    title: "Click for meeting info",
                    content: contentWindow,
                });

                google.maps.event.addListener(marker, 'click', function() {
                    infowindow.setContent(this.content);
                    infowindow.open(map, this);
                });
            }
        }
